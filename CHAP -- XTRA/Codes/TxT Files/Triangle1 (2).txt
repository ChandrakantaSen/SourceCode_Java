class Triangle1
     {
         public static void main(String args[])
               {
                   int i,j,k=0;
                   for (i=9;i<=1;i=i-2)
                      {
                          for(j=1;j<=k;j++)
                       System.out.print(" ");
                          for(j=i;j>=1;j--)
                          System.out.print("*");
                          k++;
                          System.out.println();
                        }
                    
                    }
                }
